class Stem(nn.Module):
    def __init__(self, in_channels):
        super(Stem, self).__init__()
        self.branch_0 = nn.Sequential(
            nn.Conv2d(in_channels, 32, 3, stride=2, padding=0, bias=False),
            nn.Conv2d(32, 32, 3, stride=1, padding=0, bias=False),
            nn.Conv2d(32, 64, 3, stride=1, padding=0, bias=False)
        )

        self.branch_1 = nn.MaxPool2d(3, stride=2, padding=0)
        self.branch_2 = nn.Conv2d(64, 96, 3, stride=2, padding=0, bias=False)

        self.branch_3 = nn.Sequential(
            nn.Conv2d(160, 64, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(64, 96, 3, stride=1, padding=0, bias=False),
        )
        self.branch_4 = nn.Sequential(
            nn.Conv2d(160, 64, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(64, 64, (7, 1), stride=1, padding=(3,0), bias=False),
            nn.Conv2d(64, 64, (1, 7), stride=1, padding=(0, 3), bias=False),
            nn.Conv2d(64, 96, 3, stride=1, padding=0, bias=False)
        )

        self.branch_5 = nn.Conv2d(192, 192, 3, stride=2, padding=0, bias=False)
        self.branch_6 = nn.MaxPool2d(3, stride=2, padding=0)

    def forward(self, x):
        x = self.branch_0(x)
        x0 = self.branch_1(x)
        x1 = self.branch_2(x)
        x = torch.cat((x0, x1), dim=1)
        x0 = self.branch_3(x)
        x1 = self.branch_4(x)
        x = torch.cat((x0, x1), dim=1)
        x0 = self.branch_5(x)
        x1 = self.branch_6(x)
        return torch.cat((x0, x1), dim=1)

class Inception_A(nn.Module):
    def __init__(self, in_channels):
        super(Inception_A, self).__init__()
        self.branch_0 = nn.Conv2d(in_channels, 96, 1, stride=1, padding=0, bias=False)
        self.branch_1 = nn.Sequential(
            nn.Conv2d(in_channels, 64, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(64, 96, 3, stride=1, padding=1, bias=False)
        )
        self.branch_2 = nn.Sequential(
            nn.Conv2d(in_channels, 64, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(64, 96, 3, stride=1, padding=1, bias=False),
            nn.Conv2d(96, 96, 3, stride=1, padding=1, bias=False)
        )
        self.branch_3 = nn.Sequential(
            nn.AvgPool2d(3, 1, padding=1, count_include_pad=False),
            nn.Conv2d(384, 96, 1, stride=1, padding=0, bias=True)
        )
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        x0 = self.branch_0(x)
        x1 = self.branch_1(x)
        x2 = self.branch_2(x)
        x3 = self.branch_3(x)
        _x = torch.cat((x0, x1, x2, x3), dim=1)
        return self.relu(_x)
    

class Reduction_A(nn.Module):
    def __init__(self, in_channels):
        super(Reduction_A, self).__init__()
        self.branch0 = nn.MaxPool2d(3, stride=2, padding=0)

        self.branch1 = nn.Conv2d(in_channels, 384, 3, stride=2, padding=0)

        self.branch2 = nn.Sequential(
            nn.Conv2d(in_channels, 192, 1, stride=1, padding=0),
            nn.Conv2d(192, 224, 3, stride=1, padding=1),
            nn.Conv2d(224, 256, 3, stride=2, padding=0)
        )

    def forward(self, x):
        x0 = self.branch0(x)
        x1 = self.branch1(x)
        x2 = self.branch2(x)
        return torch.cat((x0, x1, x2), 1)


class Inception_B(nn.Module):
    def __init__(self, in_channels):
        super(Inception_B, self).__init__()
        self.branch_0 = nn.Conv2d(in_channels, 384, 1, stride=1, padding=0, bias=False)
        self.branch_1 = nn.Sequential(
            nn.Conv2d(in_channels, 192, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(192, 224, (1, 7), stride=1, padding=(0, 3), bias=False),
            nn.Conv2d(224, 256, (7, 1), stride=1, padding=(3, 0), bias=False)
        )
        self.branch_2 = nn.Sequential(
            nn.Conv2d(in_channels, 192, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(192, 192, (7, 1), stride=1, padding=(3, 0), bias=False),
            nn.Conv2d(192, 224, (1, 7), stride=1, padding=(0, 3), bias=False),
            nn.Conv2d(224, 224, (7, 1), stride=1, padding=(3, 0), bias=False),
            nn.Conv2d(224, 256, (1, 7), stride=1, padding=(0, 3), bias=False)
        )
        self.branch_3 = nn.Sequential(
            nn.AvgPool2d(3, stride=1, padding=1, count_include_pad=False),
            nn.Conv2d(in_channels, 128, 1, stride=1, padding=0, bias=False)
        )
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        x0 = self.branch_0(x)
        x1 = self.branch_1(x)
        x2 = self.branch_2(x)
        x3 = self.branch_3(x)
        _x = torch.cat((x0, x1, x2, x3), dim=1)
        return self.relu(_x)


class Reduction_B(nn.Module):
    def __init__(self, in_channels):
        super(Reduction_B, self).__init__()
        self.branch_0 = nn.MaxPool2d(3, stride=2, padding=0)

        self.branch_1 = nn.Sequential(
            nn.Conv2d(in_channels, 192, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(192, 192, 3, stride=2, padding=0, bias=False),
        )
        self.branch_2 = nn.Sequential(
            nn.Conv2d(in_channels, 256, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(256, 256, (1, 7), stride=1, padding=(0, 3), bias=False),
            nn.Conv2d(256, 320, (7, 1), stride=1, padding=(3, 0), bias=False),
            nn.Conv2d(320, 320, 3, stride=2, padding=0, bias=False)
        )

    def forward(self, x):
        x0 = self.branch_0(x)
        x1 = self.branch_1(x)
        x2 = self.branch_2(x)
        return torch.cat((x0, x1, x2), dim=1)


class Inception_C(nn.Module):
    def __init__(self, in_channels, activation = True):
        super(Inception_C, self).__init__()
        self.branch_0 = nn.Conv2d(in_channels, 256, 1, stride=1, padding=0, bias=False)

        self.branch_1 = nn.Conv2d(in_channels, 384, 1, stride=1, padding=0, bias=False)
        self.branch_1_1 = nn.Conv2d(384, 256, (1, 3), stride=1, padding=(0, 1), bias=False)
        self.branch_1_2 = nn.Conv2d(384, 256, (3, 1), stride=1, padding=(1, 0), bias=False)

        self.branch_2 = nn.Sequential(
            nn.Conv2d(in_channels, 384, 1, stride=1, padding=0, bias=False),
            nn.Conv2d(384, 448, (3, 1), stride=1, padding=(1, 0), bias=False),
            nn.Conv2d(448, 512, (1, 3), stride=1, padding=(0, 1), bias=False),
        )
        self.branch_2_1 = nn.Conv2d(512, 256, (1, 3), stride=1, padding=(0, 1), bias=False)
        self.branch_2_2 = nn.Conv2d(512, 256, (3, 1), stride=1, padding=(1, 0), bias=False)

        self.branch_3 = nn.Sequential(
            nn.AvgPool2d(3, stride=1, padding=1, count_include_pad=False),
            nn.Conv2d(in_channels, 256, 1, stride=1, padding=0, bias=False)
        )
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        x0 = self.branch_0(x)
        x1 = self.branch_1(x)
        x1_1 = self.branch_1_1(x1)
        x1_2 = self.branch_1_2(x1)
        x1 = torch.cat((x1_1, x1_2), 1)
        x2 = self.branch_2(x)
        x2_1 = self.branch_2_1(x2)
        x2_2 = self.branch_2_2(x2)
        x2 = torch.cat((x2_1, x2_2), dim=1)
        x3 = self.branch_3(x)
        _x = torch.cat((x0, x1, x2, x3), dim=1)
        return self.relu(_x)

class Inceptionv4(nn.Module):
    def __init__(self, in_channels=3, classes=8):
        super(Inceptionv4, self).__init__()

        self.stem = Stem(in_channels)

        self.inceptionA = Inception_A(384)
        self.reductionA = Reduction_A(384)

        self.inceptionB = Inception_B(1024)
        self.reductionB = Reduction_B(1024)

        self.inceptionC = Inception_C(1536)

        self.average_pooling = nn.AdaptiveAvgPool2d((1, 1))
        self.dropout = nn.Dropout(0.8)
        self.linear = nn.Linear(1536, classes)

    def forward(self, x):
        x = self.stem(x)

        for i in range(5):
            x = self.inceptionA(x)
        x = self.reductionA(x)

        for i in range(10):
            x = self.inceptionB(x)
        x = self.reductionB(x)

        for i in range(5):
            x = self.inceptionC(x)

        x = self.average_pooling(x)
        x = self.dropout(x)
        x = torch.flatten(x,1)
        x = self.linear(x)
        return x

def train(model, batch, optimizer):
    model.train()
    optimizer.zero_grad()
    inputs, labels = batch
    inputs = inputs.cuda()
    labels = labels.cuda()
    outputs, aux1, aux2 = model(inputs)
    l = criterion(outputs, labels) + 0.3 * criterion(aux1, labels) + 0.3 * criterion(aux2, labels)
    l.backward()
    optimizer.step()
    return l


def evaluate(model, batch, optimizer):
    model.eval()
    inputs, labels = batch
    inputs = inputs.cuda()
    labels = labels.cuda()
    outputs, _, _ = model(inputs)
    a = accuracy(outputs, labels)
    l = criterion(outputs, labels)
    return a, l

def accuracy(outputs, labels):
    _, preds = torch.max(F.log_softmax(outputs, dim=1), dim=1)
    return torch.tensor(torch.sum(preds == labels).item() / len(preds))
  
def predicts(outputs):
    _, preds = torch.max(F.log_softmax(outputs, dim=1), dim=1)
    return preds

LEARNING_RATE = 0.00005
optimizer = torch.optim.Adam(model.parameters(), lr=LEARNING_RATE)
epoches = 5

train_loader = DataLoader(train_dataset, batch_size=500, shuffle=True)
val_loader = DataLoader(val_dataset, batch_size=500)

for epoch in range(epoches):
    print('Epoch:', epoch + 1)
    c = 0
    for batch in train_loader:
        l = train(model, batch, optimizer)
        c += len(batch[0])
        print("[%d/%d] Loss: %.2f" % (c, len(train_dataset), l.item()))
      
    for batch in train_loader:
        a, l = evaluate(model, batch, optimizer)
        print("Train Loss: %.2f, Train Accuracy: %.2f" % (l.item(), a.item()))
        break

    acc = 0
    loo = 0
    d = 0
    for batch in val_loader:
        a, l = evaluate(model, batch, optimizer)
        acc += a.item()
        loo += l.item()
        d += 1
    print("Val Loss: %.2f, Val Accuracy: %.2f" % (loo / d, acc / d))
